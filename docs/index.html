Quick Start

Get up and running in minutes

# 1. Track an LLM call with automatic evaluation
class CustomerSupportController < ApplicationController
  include PromptTracker::Trackable

  def generate_greeting
    result = track_llm_call(
      "customer_support_greeting",
      variables: { customer_name: params[:name] }
      # provider/model optional - uses version's model_config
    ) do |prompt|
      # Return just the text (simplest)
      "Hello #{params[:name]}! How can I help you today?"

      # Or return structured hash with token counts:
      # response = OpenAI::Client.new.chat(...)
      # {
      #   text: response.dig("choices", 0, "message", "content"),
      #   tokens_prompt: response.dig("usage", "prompt_tokens"),
      #   tokens_completion: response.dig("usage", "completion_tokens")
      # }
    end

    # Auto-evaluation runs automatically!

    render json: { greeting: result[:response_text] }
  end
end
# 2. Configure evaluators for your prompt
prompt = PromptTracker::Prompt.find_by(name: "customer_support_greeting")

# Length check (15% weight)
prompt.evaluator_configs.create!(
  evaluator_key: :length_check,
  weight: 0.15,
  run_mode: "sync",
  config: { min_length: 50, max_length: 500 }
)

# Keyword check (30% weight)
prompt.evaluator_configs.create!(
  evaluator_key: :keyword_check,
  weight: 0.30,
  run_mode: "sync",
  config: { required_keywords: ["hello", "help"] }
)

# LLM judge (55% weight, only runs if keyword check passes)
prompt.evaluator_configs.create!(
  evaluator_key: :gpt4_judge,
  weight: 0.55,
  run_mode: "async",
  depends_on: "keyword_check",
  min_dependency_score: 80
)

Installation

Add to your Rails 7.2+ application

gem "prompt_tracker" 
# Install and setup
$ bundle install
$ bin/rails prompt_tracker:install:migrations
$ bin/rails db:migrate

# Mount the engine in config/routes.rb
mount PromptTracker::Engine => "/prompt_tracker"

# Start the server
$ bin/rails server

# Visit the dashboard
# http://localhost:3000/prompt_tracker
